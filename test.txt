global	_start
	section	.text

_start:
	call func

	; exit(result)
	mov rdi, 1
	mov	rax,60			; exit(2)
	syscall

func:
	push rbp
	mov rbp, rsp

	sub rsp, 2
	mov [rsp], byte 'H'
	mov [rsp+1], byte 'i'

	mov	rdi, 1			; fd
	mov	rsi, rsp		   ; buffer
	mov	rdx, 4 			; count
	mov	rax, 1	 		; write(2)
	syscall
	
	mov rsp, rbp
	pop rbp
	ret
